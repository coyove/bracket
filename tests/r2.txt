function foo1(a, b, c)
return a, b, c
end

assert(a == 1 and b + c == 5)
a,b,c = foo1(c=10,b="a")

function foo1(a, b, c, ...)
return a, b, c, ...
end

local _, _, _, x = foo1(1,2,3,"4")
assert(x == "4")

_, b, _, x = foo1(["b"]=2)
assert(b == 2 and x == nil)
