var n = 0 
var nCPU = 1 

fun A(i, j) {
    return ((i+j)*(i+j+1)/2 + i + 1) 
}

var K = std.opcode.closure, OP= std.opcode 
var A2 = K.empty() 
var call = (fun(){
    return fun() {
        var stack = copy() 
        var x = K[stack[0]] 
        stack[0] = A2 
        std.apply(x, stack) 
    } 
})() 

call("set", "argscount", 2) 
call("set", "envescaped", 0) 

call("writeconst", 1) 
call("writeconst", 2)

// A2 can save 3 opcodes, 10% faster than A

fun TimesVec(v, s, n, u, wg) {
    var ul = len(u) 
    for i = s  ,n {
        var vi = 0 
        for j = 0,ul {
            vi = vi + u[j] / A2(i, j) 
        }
        v[i] = vi 
    }
    wg.done() 
}

fun TimesTranspVec(v, s, n, u, wg) {
    var ul = len(u) 
    for i = s, n  {
        var vi = 0 
        for j = 0, ul  {
            vi = vi + u[j] / A2(j, i) 
        }
        v[i] = vi 
    }
    wg.done() 
}

fun ATimesTransp(v, u) {
    var x = std.genlist(len(u)) 
    var wg = std.sync.waitgroup() 

    for i = 0, nCPU {
        wg.add(1) 
        std.sync.run(TimesVec, x, i*len(v)/nCPU, (i+1)*len(v)/nCPU, u, wg) 
    }
    wg.wait() 

    for  i = 0, nCPU {
        wg.add(1) 
        std.sync.run(TimesTranspVec, v, i*len(v)/nCPU, (i+1)*len(v)/nCPU, x, wg) 
    }
    wg.wait() 
}

n = 550 
var u = {} 
var v = {} 
 
assert A2(1,1) == A(1,1) 

for idx = 0, n {
    u[idx] = 1 
    v[idx] = 0 
}

for i = 0, 10 {
    ATimesTransp(v, u) 
    ATimesTransp(u, v) 
}

var vBv = 0, vv = 0 
copy(v, fun(i, x) {
    vBv = vBv + u[i] * x 
    vv = vv + x * x 
})
