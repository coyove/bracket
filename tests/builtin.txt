fun var sum() {
    var x = arguments
    var s = 0 
    if typeof x[0] == "str" { 
        s = "" 
    }

    copy(x, fun(i, n) {s = s+n }) 
    return s 
}

assert sum(1, -1) == 0 
assert sum(1, 2, 3) == 6  
assert sum("1", "2") == "12"
